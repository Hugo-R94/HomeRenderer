struct Vec2 {
    float x, y;
};

class Geo3D {
private:
    // Données essentielles d'un .obj
    std::vector<Vec3>       _vertices;   // Positions (v)
    std::vector<Vec3>       _normals;    // Normales (vn)
    std::vector<Vec2>       _texcoords;  // Coordonnées texture (vt)
    std::vector<uint32_t>   _indices;    // Indices des faces (f)

public:
    Geo3D();
    ~Geo3D();
    
    // Getters
    const std::vector<Vec3>&     getVertices() const;
    const std::vector<Vec3>&     getNormals() const;
    const std::vector<Vec2>&     getTexcoords() const;
    const std::vector<uint32_t>& getIndices() const;
    
    // Setters
    void setVertices(const std::vector<Vec3>& vertices);
    void setNormals(const std::vector<Vec3>& normals);
    void setTexcoords(const std::vector<Vec2>& texcoords);
    void setIndices(const std::vector<uint32_t>& indices);
    
    // Chargement
    bool loadFromFile(const std::string& filepath);
    void clear();
};
